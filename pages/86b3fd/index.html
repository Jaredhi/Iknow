linux-安装脚本

# JDK

提示

此脚本适合第一次安装JDK，如果不是，请先清除环境变量

- 清除环境变量
# 查看环境变量（看这两个 PATH、JAVA_HOME）
env
# PATH 修改为默认
#(注意看你的 PATH 是否有其他环境，只需删了jdk那一段即可)
export PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/root/bin::/root/bin::

  1. 安装包下载地址：

  2. 将下面内容复制到 JDK-install.sh ，执行sh JDK-install.sh

JDK-install.sh 
#!/bin/sh
#---------脚本变量---------
# jdk 压缩包路径
FILEPATH=/home/software
# 包名
JDK8=jdk-8u161_linux-x64.tar.gz
JDK11=jdk-11_linux-x64_bin.tar.gz
# 解压缩路径、环境变量路径
TARGET=/usr/local/JDK
#---------脚本变量---------

SELECT(){
	read -p " 请输入jdk8 或 jdk11进行安装 : " tmp
    if [ ${tmp} = 'jdk8' ];then
        echo -e "\033[33m 正在为您校验jdk8安装包... \033[0m"
        CHECK8
    elif [ ${tmp} = 'jdk11' ];then
        echo -e "\033[33m 正在为您校验jdk8安装包... \033[0m"
        CHECK11
    exit
    fi
}


CHECK8(){
if [ -f "${FILEPATH}/${JDK8}" ]; then
echo -e "\033[32m 校验成功！！ \033[0m"
INSTALL8_ING
else
echo -e "\033[31m 校验失败！！请上传jdk安装包至'${FILEPATH}'文件夹下 \033[0m"
	exit
fi
}
CHECK11()
{
if [ -f "${FILEPATH}/${JDK11}" ]; then
echo -e "\033[32m 校验成功！！ \033[0m"
INSTALL11_ING
else
echo -e "\033[31m 校验失败！！请上传jdk安装包至'${FILEPATH}'文件夹下 \033[0m"
	exit
fi
}

PRESS_INSTALL()
{
    echo -e "\033[32m 即将安装JDK... \033[0m"
	echo -e "\033[32m Press any key to install...or Press Ctrl+c to cancel \033[0m"
    SAVE=`stty -g`
    stty -icanon -echo min 1 time 0
    dd count=1 2>/dev/null
    stty ${SAVE}
}

INSTALL8_ING()
{

PRESS_INSTALL

echo -e "\033[32m 开始安装JDK8... \033[0m"
mkdir ${TARGET}
tar -zxvf ${FILEPATH}/${JDK8} -C ${TARGET}

echo "#JAVA_8_HOME">>/etc/profile
echo "export JAVA_HOME="${TARGET}"/jdk1.8.0_161">>/etc/profile
source /etc/profile
echo "export CLASSPATH=.:\$JAVA_HOME/lib/dt.jar:\$JAVA_HOME/lib/tools.jar">>/etc/profile
echo "export PATH=\$JAVA_HOME/bin:\$PATH">>/etc/profile
source /etc/profile
java -version

if [ $? -eq 0 ]; then
	echo -e "\033[32m 安装成功！！！ \033[0m"

	else
	echo -e "\033[031m 安装失败！！！ \033[0m"
   exit
fi
}

INSTALL11_ING()
{

PRESS_INSTALL

echo -e "\033[32m 开始安装JDK11... \033[0m"
mkdir ${TARGET}
tar -zxvf ${FILEPATH}/${JDK11} -C ${TARGET}

echo "#JAVA_HOME">>/etc/profile
echo "export JAVA_HOME="${TARGET}"/jdk-11">>/etc/profile
echo "export PATH=\$JAVA_HOME/bin:\$PATH:">>/etc/profile

source /etc/profile
java -version

if [ $? -eq 0 ]; then
	echo -e "\033[32m 安装成功！！！ \033[0m"

	else
	echo -e "\033[031m 安装失败！！！ \033[0m"
   exit
fi
}

###########################卸载##########################

#输入n
STOP_UNINSTALL()
{
    echo -e "\033[33m 已停止卸载!! \033[0m"
	exit
}

#输入错误
UNINSTALL_AGAIN()
{
  echo -e "\033[33m 请重新输入!!! \033[0m"
  UNINSTALL_CHOOSE
}

#输入y
UNINSTALL_ING()
{
 killall -9 java

 rm -rf ${TARGET}
 java -version
 
if [ $? -eq 0 ]; then

	echo -e "\033[031m 卸载失败，请手动用rm命令卸载!! \033[0m"
    exit	
else
	echo -e "\033[32m 卸载成功！！！ \033[0m"
	echo -e "\033[32m 请手动清除 /etc/profile -> JAVA_HOME 的环境变量 \033[0m"
fi
}

#开始输入
UNINSTALL_CHOOSE()
{
 read -p "是否卸载JDK？yes/no: "  Select

 case "${Select}" in
    y|Y|yes)
        Select="y"
		UNINSTALL_ING
    ;;
    n|N|no)
	    Select="n"
		STOP_UNINSTALL
    ;;
	*)
        echo "输入错误"
		UNINSTALL_AGAIN
    esac
	

}

###########################卸载###########################


JDK=`rpm -qa | grep jdk`
echo -e "\033[33m 开始检测本机是否安装JDK... \033[0m"
VS=`java -version`
if [ $? -eq 0 ]; then
    echo -e "\033[33m 检测到您已安装JDK \033[0m"
	UNINSTALL_CHOOSE
    
else
    echo -e "\033[33m 检测到您尚未安装JDK \033[0m"
    SELECT
    source /etc/profile
fi
